async def parse_send_message_response(resp):
    data = await resp.json()
    if not data.get('ok'):
        err = {
            'ok': False,
            'error_code': data.get('error_code'),
            'description': data.get('description'),
            'parameters': data.get('parameters', {})  # {retry_after, migrate_to_chat_id}
        }
        return err
    r = data['result']
    return {
        'ok': True,
        'message_id': r.get('message_id', 0),
        'chat_id': (r.get('chat') or {}).get('id'),
        'date': r.get('date'),
        'text': r.get('text') or r.get('caption'),
    }
